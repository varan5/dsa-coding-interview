# Time: O(n^2)	  Space: O(1)

def two_sum(array, target_sum):
    for i in range(len(array) -1):
    	for j in range(i, len(array)):
	    if array[i] + array[j] == target_sum:
	    	return [i, j]	# return the index of the found numbers
    
    return []	# pair of two sum not found, so return empty list

def two_sum(array, target_sum):
    start = 0
    end = len(array) -1
    while start < end:
    	potential_sum = array[start] + array[end]
	if potential_sum == target_sum:
	    return [start, end]    # return the index of both found elements whose sum is equal to the target sum
	elif potential_sum < target_sum:
	    start += 1
	elif potential_sum > target_sum:
